API Documentation

Endpoint: /add_garam
Description

This endpoint is used to add salt quality data to the database.

Request Method

POST

Request URL

https://cekgaram.aisoru.id/add_garam

Request Body
Field Type Description
id_petani Number ID of the farmer
kadar_air Number Moisture content
kadar_garam Number Salt content
Example Usage
Python
import requests

url = "https://cekgaram.aisoru.id/add_garam"
data = {
    "id_petani": {{ user_data[0] }},
    "kadar_air": 10.5,
    "kadar_garam": 15.3
}

response = requests.post(url, json=data)
print(response.text)

Java
URL url = new URL("https://cekgaram.aisoru.id/add_garam");
HttpURLConnection con = (HttpURLConnection) url.openConnection();
con.setRequestMethod("POST");
con.setRequestProperty("Content-Type", "application/json; utf-8");
con.setRequestProperty("Accept", "application/json");
con.setDoOutput(true);

String jsonInputString = "{\"id_petani\": {{ user_data[0] }}, \"kadar_air\": 10.5, \"kadar_garam\": 15.3}";

try(OutputStream os = con.getOutputStream()) {
    byte[] input = jsonInputString.getBytes("utf-8");
    os.write(input, 0, input.length);           
}

try(BufferedReader br = new BufferedReader(new InputStreamReader(con.getInputStream(), "utf-8"))) {
    StringBuilder response = new StringBuilder();
    String responseLine = null;
    while ((responseLine = br.readLine()) != null) {
        response.append(responseLine.trim());
    }
    System.out.println(response.toString());
}
